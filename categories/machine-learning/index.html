线性回归，岭回归和Lasso回归

#

线性回归

线性回归的一般形式

$$
\tag{x是向量}假设函数：h_\theta(x) = \theta_0x_0 + \theta_1x_1 + \dots +\theta_nx_n=\theta^Tx
\qquad
$$

$$
损失函数：J(\theta) = {1 \over 2m}\sum_{i=1}^m\Big({h_\theta\big({x^{(i)}}\big)-y^{(i)}}\Big)^2
$$

$$
目标: 求minJ(\theta_0,\theta_1,\dots,\theta_n)时的\theta 值
$$

梯度下降法求解

$$
\begin{alignedat}{2}
\frac{\partial}{\partial\theta_j}J(\theta)=
\frac{\partial}{\partial\theta_j}\frac{1}{2m}\sum_{i=1}^m\Big({h_\theta\big({x^{(i)}}\big)-y^{(i)}}\Big)^2\quad\
=\frac{1}{m}\sum_{i=1}^m\Big(h_\theta\big(x^{(i)}\big)-y^{(i)}\Big)
\frac{\partial}{\partial\theta_j}(\theta^Tx)\
=\frac{1}{m}\sum_{i=1}^m\Big(h_\theta\big(x^{(i)}\big)-y^{(i)}\Big)\cdot{x_j^{(i)}}\qquad

\end{alignedat}
$$

$$
\theta_j := \theta_j - \alpha\frac{\partial}{\partial\theta_j}J(\theta) \qquad (\alpha{为学习率})\qquad\qquad\qquad\
=\theta_j - \frac{\alpha}{m}\sum_{i=1}^m\Big(h_\theta\big(x^{(i)}\big)-y^{(i)}\Big)\cdot{x_j^{(i)}} \qquad(1)
$$

向量化上述公式

$$
\begin{equation}
\begin{split}

&X = \begin{vmatrix}
x^{(1)}\
x^{(2)}\
\dots\
x^{(m)}
\end{vmatrix}
=\begin{vmatrix}
x_0^{(1)} & x_1^{(1)} & \dots & x_n^{(1)} \
x_0^{(2)} & x_1^{(2)} & \dots & x_n^{(2)} \
\dots & \dots & \dots & \dots & \
x_0^{(m)} & x_1^{(m)} & \dots & x_n^{(m)}
\end{vmatrix}\
&\

&Y=\begin{vmatrix}
y^{(1)}\
y^{(2)}\
\dots \
y^{(m)}
\end{vmatrix} \qquad

\theta=\begin{vmatrix}
\theta^{(0)}\
\theta^{(1)}\
\dots \
\theta^{(n)}
\end{vmatrix}\
&\

&先求x\cdot\theta并记为A:
A=X\cdot\theta=
\begin{vmatrix}
x_0^{(1)} & x_1^{(1)} & \dots & x_n^{(1)} \
x_0^{(2)} & x_1^{(2)} & \dots & x_n^{(2)} \
\dots & \dots & \dots & \dots & \
x_0^{(m)} & x_1^{(m)} & \dots & x_n^{(m)}
\end{vmatrix} \cdot
\begin{vmatrix}
\theta^{(0)}\
\theta^{(1)}\
\dots \
\theta^{(n)}
\end{vmatrix}
=\begin{vmatrix}
\theta_0x_0^{(1)}+\theta_1x_1^{(1)}+\dots+\theta_nx_n^{(1)} \
\theta_0x_0^{(2)}+\theta_1x_1^{(2)}+\dots+\theta_nx_n^{(2)} \
\dots \
\theta_0x_0^{(m)}+\theta_1x_1^{(m)}+\dots+\theta_nx_n^{(m)} \
\end{vmatrix}\
&\

&求h\theta(x)-y并记为E:\
&E = h_\theta(x)-Y=
\begin{vmatrix}
g(A^{(1)}) - y^{(1)} \
g(A^{(2)}) - y^{(2)} \
\dots \
g(A^{(m)}) - y^{(m)} \
\end{vmatrix}=
\begin{vmatrix}
e^{(1)} \
e^{(2)} \
\dots \
e^{(m)}
\end{vmatrix}
=g(A) - Y \
&\

\end{split}
\end{equation}
$$

​
$$
\begin{equation}
\begin{split}

&\color{red}把上述公式代入公式(1)中 \
&\
&\theta_0 := \theta_0 - \frac{\alpha}{m}\sum_{i=1}^m\Big(h_\theta\big(x^{(i)}\big)-y^{(i)}\Big)\cdot{x_0^{(i)}} \
&=\theta_0 - \frac{\alpha}{m}\sum_{i=1}^me^{(i)}x_0^{(i)} \
&=\theta_0 - \frac{\alpha}{m}\cdot(x_0^{(1)},x_0^{(2)},\dots,x_0^{(m)})\cdot{E} \
&\

&同样的可以写出\theta_j, \
&\theta_j :=\theta_j - \frac{\alpha}{m}\cdot(x_j^{(1)},x_j^{(2)},\dots,x_j^{(m)})\cdot{E}\
&\

&综合起来 \
&\begin{vmatrix}
\theta^{(0)}\
\theta^{(1)}\
\dots \
\theta^{(n)}
\end{vmatrix}
:=\begin{vmatrix}
\theta^{(0)}\
\theta^{(1)}\
\dots \
\theta^{(n)}
\end{vmatrix}
-\frac{\alpha}{m}\cdot
\begin{vmatrix}
x_0^{(1)},x_0^{(2)},\dots,x_0^{(m)} \
x_1^{(1)},x_1^{(2)},\dots,x_1^{(m)} \
\dots \
x_n^{(1)},x_n^{(2)},\dots,x_n^{(m)}
\end{vmatrix}
\cdot{E}
=\theta - \frac{\alpha}{m}\cdot{X^T}\cdot{E}
\qquad \qquad \qquad \qquad \qquad \qquad \qquad \qquad

\end{split}
\end{equation}
$$

$$
\begin{equation}
\begin{split}
&综上所述，vectorization后\theta更新的步骤 \
&(1) 求A=X\cdot\theta \
&(2) 求E=g(A)-Y \
&(3) 求\theta:=\theta-\alpha\cdot{X}^T\cdot{E} \
&综合起来 \
&\theta := \theta-\alpha\cdot(\frac{1}{m})\cdot{X^T}\cdot\Big(g(X\cdot\theta)-Y\Big)\
&\quad=\theta - \frac{\alpha}{m}\cdot{X^T}\cdot(X\theta-Y)

\end{split}
\end{equation}
$$

正规方程求解

$$
\color{black} \large 通过求解\frac{\partial}{\partial\theta_j}J(\theta)=0的方式来求出使得损失函数最小情况下为0时的\theta值
$$

$$
\begin{equation}
\begin{split}

&\frac{\partial}{\partial\theta_j}J(\theta)=0 \
&\
&J(\theta) = \frac{1}{2m}(X\cdot\theta-Y)^T(X\cdot\theta-Y) \
&\qquad=\frac{1}{2m}(Y^TY-Y^TX\theta-\theta^TX^TY+\theta^Tx^TX\theta)\
&\
&对J(\theta)进行求导 \
&\
&\frac{\partial}{\partial\theta_j}J(\theta)=\frac{1}{2m}(\frac{\partial{Y^T}Y}{\partial\theta}-\frac{\partial{Y^T}X\theta}{\partial\theta}-\frac{\partial{\theta^TX^T}Y}{\partial\theta}-\frac{\partial\theta^TX^TX\theta}{\partial\theta})\
&\
&\large\color{red}求导公式:\
&\
&\frac{dAB}{dB} = A^T; \qquad \frac{dX^TAX}{dX}=2AX \
&\
&(求导第一项)\
&\
&\qquad\frac{\partial{Y^T}Y}{\partial\theta}=0 \
&\
&(求导第二项)\
&\qquad Y^TX\theta=\begin{vmatrix}y^{(1)},y^{(2)},\dots,y^{(m)}\end{vmatrix}
\begin{vmatrix}
x_0^{(1)} & x_1^{(1)} & \dots & x_n^{(1)} \
x_0^{(2)} & x_1^{(2)} & \dots & x_n^{(2)} \
\dots & \dots & \dots & \dots & \
x_0^{(m)} & x_1^{(m)} & \dots & x_n^{(m)}
\end{vmatrix}
\begin{vmatrix}
\theta^{(0)},\theta^{(1)}, \dots, \theta^{(n)}
\end{vmatrix}^T\
&\
&\qquad\qquad\quad=(x_0^{(1)}y^{(1)}+\dots+x_0^{(m)}y^{(m)})\theta_0 + (x_1{(1)}y^{(1)}+\dots+x_1{(m)}y^{(m)})\theta_1 + \dots + (x_n{(1)}y^{(1)}+\dots+x_n{(m)}y^{(m)})\theta_n\
&\
&\qquad\frac{\partial{Y^TX\theta}}{\partial\theta}
=\begin{vmatrix}
\frac{\partial{Y^TX\theta}}{\partial\theta_0} \
\dots \
\frac{\partial{Y^TX\theta}}{\partial\theta_n}
\end{vmatrix}
=\begin{vmatrix}
x_0^{(1)}y^{(1)}+\dots+x_0^{(m)}y^{(m)} \
\dots \
x_n^{(1)}y^{(1)}+\dots+x_n^{(m)}y^{(m)}
\end{vmatrix}
=X^TY \
&\
&(求导第三项)\
& \
&\qquad\theta^TX^TY=
\begin{vmatrix}
\theta^{(0)}, \theta^{(1)}, \dots, \theta^{(n)}
\end{vmatrix}
\begin{vmatrix}
x_0^{(1)} & x_1^{(1)} & \dots & x_n^{(1)} \
x_0^{(2)} & x_1^{(2)} & \dots & x_n^{(2)} \
\dots & \dots & \dots & \dots & \
x_0^{(m)} & x_1^{(m)} & \dots & x_n^{(m)}
\end{vmatrix}^T
\begin{vmatrix}y^{(1)},y^{(2)},\dots,y^{(m)}\end{vmatrix}^T \
& \
&\qquad\qquad\quad=
(x_0^{(1)}\theta_0+\dots+x_n^{(1)}\theta_n)y^{(1)} + (x_0^{(2)}\theta_0+\dots+x_n^{(2)}\theta_n)y^{(2)} + \dots +(x_0^{(m)}\theta_0+\dots+x_n^{(m)}\theta_n)y^{(m)} \
&\
&该矩阵求导为分母布局下的标量/向量形式：\
&\
&\qquad\frac{\partial\theta^TX^TY}{\partial\theta}
=\begin{vmatrix}
\frac{\partial\theta^TX^TY}{\partial\theta_0} \
\dots \
\frac{\partial\theta^TX^TY}{\partial\theta_n}
\end{vmatrix}
=\begin{vmatrix}
x_0^{(1)}y^{(1)}+\dots+x_0^{(m)}y^{(m)} \
\dots \
x_n^{(1)}y^{(1)}+\dots+x_n^{(m)}y^{(m)}
\end{vmatrix}
=X^TY \
&\
&(求导第四项)\
& \
&\qquad \theta^TX^TX\theta = (X^TX)(\theta_0^2 + \theta_1^2 + \dots + \theta_n^2) \
& \
& 其中\theta^TX^TX\theta为标量，可看成一个常数，矩阵求导为分母布局下的标量/向量形式：\
& \
&\qquad \frac{\partial\theta^TX^TX\theta}{\partial\theta}
= \begin{vmatrix}
\frac{\partial\theta^TX^TX\theta}{\partial\theta_0} \
\dots \
\frac{\partial\theta^TX^TX\theta}{\partial\theta_n}
\end{vmatrix}
= 2(X^TX)
\begin{vmatrix}
\theta^{(0)}\
\theta^{(1)}\
\dots \
\theta^{(n)}
\end{vmatrix}
=2X^TX\theta \
&\
&综上，正规方程为：\
& \
&\qquad \frac{1}{2m}(-2X^TY + 2X^TX\theta) = 0
&\
&特征参数表示如下：\
&\
&\qquad \theta=(X^TX)^{-1}X^Ty=0 \qquad(X为m*(n+1)矩阵) \

\end{split}
\end{equation}
$$

岭回归与Lasso回归

岭回归的一般形式

$$
\tag{x是向量}假设函数：h_\theta(x) = \theta_0x_0 + \theta_1x_1 + \dots +\theta_nx_n=\theta^Tx
$$

$$
岭回归的损失函数：J(\theta)=\frac{1}{2m}\sum_{i=1}^m\Big(h_\theta\big(x^{(i)}\big)-y^{(i)}\Big)^2 + \lambda \sum_{j=1}^n\theta_j^2 \quad
$$

$$
Lasso回归的损失函数：J(\theta)={1 \over 2m}\sum_{i=1}^m\Big(h_\theta\big(x^{(i)}\big)-y^{(i)}\Big)^2 + \lambda\sum_{j=1}^n|\theta_j|
$$

$$
目标: 求minJ(\theta_0,\theta_1,\dots,\theta_n)时的\theta 值
$$

$$
\color{red} \Big (其中\lambda为正则化参数，因为不对\theta_0进行惩罚，所以J的值是从1开始)
$$

岭回归梯度下降法

$$
\frac{\partial}{\partial\theta_j}J(\theta) = \frac{\partial}{\partial\theta_j}\cdot\frac{1}{2m}\Bigg[\sum_{i=1}^m\Big(h_\theta\big(x^{(i)}\big)-y^{(i)}\Big)^2 + \lambda \sum_{j=1}^n\theta_j^2 \Bigg]\
=\frac{1}{m}\sum_{i=1}^m\Big(h_\theta\big(x^{(i)}\big)-y^{(i)}\Big)\cdot{X_j^{(i)}} + \frac{\lambda}{m}\theta_j\tag{1}
$$

$$
\color{red} \Large \bf {因为\theta_0不参与惩罚，因而分为两部分\qquad\quad}
$$

$$
\theta_0 := \theta_0 - \alpha\sum_{i=1}^m\Big(h_\theta\big(x^{(i)}\big)-y^{(i)}\Big)X_0^{(i)}\tag{2}
$$

$$
\theta_j := \theta_j - \alpha\frac{\partial}{\partial\theta_j}J(\theta)\tag{3}
$$

$$
{把(1)带入(3)中}:\
\theta_j := \theta_j - \frac{\alpha}{m}\sum_{i=1}^m\Big(h_\theta\big(x^{(i)}\big)-y^{(i)}\Big)\cdot{X_j}^{(i)} - \frac{\alpha\lambda}{m}\theta_j\
:= (1 - \frac{\alpha\lambda}{m})\theta_j - \frac{\alpha}{m}\sum_{i=1}^m\Big(h_\theta\big(x^{(i)}\big)-y^{(i)}\Big)\cdot{X_j^{(i)}}\
若m训练样本较大时，\frac{\lambda}{m}趋向于0，则每次均会进行微调
$$

正规方程求解法

$$
\theta = \Biggr(X^TX + \lambda\begin{vmatrix}
0 & & & & \
& 1 & & & \
& & . & & \
& & & . & \
& & & & 1 \
\end{vmatrix}\Biggr)^{-1}\cdot{X^T}y
$$

可能存在的问题

$$
X^TX + \lambda\begin{vmatrix}
0 & & & & \
& 1 & & & \
& & . & & \
& & & . & \
& & & & 1 \
\end{vmatrix}可证明其不可能是奇异矩阵或者退化矩阵，解决一般线性回归正规方程解中出现的奇异矩阵
$$

知识点总结

$$
\color{blue}
\begin{equation}
\begin{split}
&1. 梯度下降法优化：特征缩放 令x_n = \frac{x_n-\mu_n}{s_n} \qquad （s_n为最大值减去最小值）\
&2. 学习率\alpha的选择可考虑为：0.01，0.03，0.1，0.3，1.3 \
&3. 绘制损失函数和迭代次数的图像来观测何时收敛 \
&4. 线性回归的衍生版，把多项式转化为线性回归的方式，然后进行特征缩放
\end{split}
\end{equation}
$$
